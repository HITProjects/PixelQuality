52   def annotate_and_build_heatmaps(images_dir: str,
53                                   json_name: str = "points_by_image.json",
54                                   sigma: int = 30,
55                                   alpha: float = 0.5,
56                                   output_suffix: str = "_HEAT.png"):
57       """ 
58       1) Iterates over all images in `images_dir` (png/jpg/jpeg/bmp). 
59       2) For each image, shows an interactive window to collect EXACTLY 3 points from the user. 
60          - Left click: add a point 
61          - Right click or key 'u': undo last point 
62          - Key 'r': reset points 
63          - Key 'q': quit early (no save for current) 
64          Once 3 points are set, the image auto-advances to the next one. 
65       3) Saves all collected points into a single JSON file in the same folder. 
66       4) After all images are annotated, generates a LocJND-style heatmap overlay for each image 
67          with sigma=30 and alpha=0.5 (configurable) and saves it next to the original. 
68       """
69   
70       # ------------- gather images -------------
71       exts = ("*.png", "*.jpg", "*.jpeg", "*.bmp")
72       image_paths = []
73       for e in exts:
74           image_paths.extend(glob(os.path.join(images_dir, e)))
75       image_paths.sort()
76       if not image_paths:
77           print("No images found in:", images_dir)
78           return
79   
80       # ------------- interactive picker helpers -------------
81       collected = []  # list of dicts: {"filename": str, "width": int, "height": int, "points": [{"x":..,"y":..}, ...]}
82   
83       max_win_w, max_win_h = 1400, 900  # display cap (image is still recorded in original resolution)
84   
85       def show_picker(img_bgr, title="Pick 3 points"):
86           """ 
87           Returns a list of 3 (x,y) pixel coordinates in ORIGINAL resolution. 
88           """
89           h, w = img_bgr.shape[:2]
90           scale = min(max_win_w / w, max_win_h / h, 1.0)
91           disp = cv2.resize(img_bgr, (int(w * scale), int(h * scale)), interpolation=cv2.INTER_AREA)
92   
93           pts = []
94   
95           def redraw():
96               view = disp.copy()
97               for i, (ox, oy) in enumerate(pts, 1):
98                   dx, dy = int(ox * scale), int(oy * scale)
99                   cv2.circle(view, (dx, dy), 8, (0, 0, 255), -1)
100                  cv2.putText(view, str(i), (dx + 10, dy - 10),
101                              cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 2)
102              footer = "L: add  |  R/u: undo  |  r: reset  |  q: quit"
103              cv2.rectangle(view, (0, view.shape[0]-30), (view.shape[1], view.shape[0]), (0,0,0), -1)
104              cv2.putText(view, footer, (10, view.shape[0]-10),
105                          cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255,255,255), 1)
106              cv2.imshow(title, view)
107  
108          def on_mouse(event, x, y, flags, param):
109              nonlocal pts
110              if event == cv2.EVENT_LBUTTONDOWN and len(pts) < 3:
111                  ox, oy = int(x / scale), int(y / scale)
112                  ox = max(0, min(w - 1, ox))
113                  oy = max(0, min(h - 1, oy))
114                  pts.append((ox, oy))
115                  redraw()
116              elif event == cv2.EVENT_RBUTTONDOWN:
117                  if pts:
118                      pts.pop()
119                      redraw()
120  
121          cv2.namedWindow(title, cv2.WINDOW_AUTOSIZE)
122          cv2.setMouseCallback(title, on_mouse)
123          redraw()
124  
125          while True:
126              k = cv2.waitKey(20) & 0xFF
127              if k == ord('q'):
128                  pts = []  # empty to signal "skip"
129                  break
130              elif k == ord('u'):
131                  if pts:
132                      pts.pop()
133                      redraw()
134              elif k == ord('r'):
135                  pts = []
136                  redraw()
137              # auto-advance when 3 points chosen
138              if len(pts) == 3:
139                  break
140  
141          cv2.destroyWindow(title)
142          return pts
143  
144      # ------------- pass 1: collect points -------------
145      for idx, img_path in enumerate(image_paths, 1):
146          img = cv2.imread(img_path)
147          if img is None:
148              print(f"Skipping unreadable file: {img_path}")
149              continue
150  
151          base = os.path.basename(img_path)
152          print(f"[{idx}/{len(image_paths)}] {base}: pick 3 points...")
153          pts = show_picker(img, title=f"Pick 3 points: {base}")
154  
155          if len(pts) != 3:
156              print(f"Skipped (no points) â†’ {base}")
157              continue
158  
159          h, w = img.shape[:2]
160          collected.append({
161              "filename": base,
162              "width": w,
163              "height": h,
164              "points": [{"x": int(x), "y": int(y)} for (x, y) in pts]
165          })
166  
167      # save JSON
168      json_path = os.path.join(images_dir, json_name)
169      with open(json_path, "w", encoding="utf-8") as f:
170          json.dump({"images": collected}, f, ensure_ascii=False, indent=2)
171      print(f"Saved annotations â†’ {json_path}")
172  
173      # ------------- pass 2: build heatmaps -------------
174      def build_heat_overlay(img_bgr, points_xy, sigma_px=30, alpha_overlay=0.6):
175          h, w = img_bgr.shape[:2]
176          heat = np.zeros((h, w), dtype=np.float32)
177          xs = np.arange(0, w)
178          ys = np.arange(0, h)
179          X, Y = np.meshgrid(xs, ys)
180  
181          for (x, y) in points_xy:
182              g = np.exp(-((X - x) ** 2 + (Y - y) ** 2) / (2.0 * (sigma_px ** 2)))
183              heat += g
184  
185          if heat.max() > 0:
186              heat /= heat.max()
187  
188          heat_u8 = (heat * 255).astype(np.uint8)
189          colored = cv2.applyColorMap(heat_u8, cv2.COLORMAP_JET)  # BGR
190          overlay = cv2.addWeighted(img_bgr, 1.0 - alpha_overlay, colored, alpha_overlay, 0)
191          return overlay
192  
193      for entry in collected:
194          img_path = os.path.join(images_dir, entry["filename"])
195          img = cv2.imread(img_path)
196          if img is None:
197              print(f"Missing image for heatmap: {img_path}")
198              continue
199  
200          pts_xy = [(p["x"], p["y"]) for p in entry["points"]]
201          overlay = build_heat_overlay(img, pts_xy, sigma_px=sigma, alpha_overlay=alpha)
202  
203          out_name = os.path.splitext(entry["filename"])[0] + output_suffix
204          out_path = os.path.join(images_dir, out_name)
205          cv2.imwrite(out_path, overlay)
206          print(f"Saved heatmap â†’ {out_path}")
207  
208      print("Done.")

213  def collect_points_from_heatmaps(heatmaps_dir: str,
214                                   json_name: str = "heatmap_points.json",
215                                   require_name_contains: str = "HEAT"):
216      """ 
217      Iterate over all heatmap images in a folder, display each, and let the user click EXACTLY 3 points. 
218      Saves all points into a single JSON file at the end. 
219   
220      Controls: 
221        - Left click: add a point (max 3) 
222        - Right click or key 'u': undo last point 
223        - Key 'r': reset current image's points 
224        - Key 'q': quit early (keeps progress so far) 
225   
226      Args: 
227          heatmaps_dir: Path to a directory containing heatmap images. 
228          json_name: Output JSON filename (saved inside heatmaps_dir). 
229          require_name_contains: Only files containing this substring are considered (set to "" to disable). 
230      """
231      # ---- collect candidate image paths ----
232      exts = ("*.png", "*.jpg", "*.jpeg", "*.bmp")
233      paths = []
234      for e in exts:
235          paths.extend(glob(os.path.join(heatmaps_dir, e)))
236      # keep only files that look like heatmaps (default: filename contains "HEAT")
237      if require_name_contains:
238          paths = [p for p in paths if require_name_contains.lower() in os.path.basename(p).lower()]
239      paths.sort()
240  
241      if not paths:
242          print("No heatmap images found in:", heatmaps_dir)
243          return
244  
245      # display size cap; coordinates are always saved in ORIGINAL resolution
246      MAX_W, MAX_H = 1400, 900
247  
248      annotations = []  # [{"filename": ..., "width": w, "height": h, "points": [{"x":..,"y":..}, ...]}]
249  
250      def pick_points(img_bgr, window_title):
251          """Return up to 3 points (x,y) in ORIGINAL pixel coordinates."""
252          h, w = img_bgr.shape[:2]
253          scale = min(MAX_W / w, MAX_H / h, 1.0)
254          disp = cv2.resize(img_bgr, (int(w * scale), int(h * scale)), interpolation=cv2.INTER_AREA)
255  
256          pts = []
257  
258          def redraw():
259              view = disp.copy()
260              for i, (ox, oy) in enumerate(pts, 1):
261                  dx, dy = int(ox * scale), int(oy * scale)
262                  cv2.circle(view, (dx, dy), 8, (0, 0, 255), -1)
263                  cv2.putText(view, str(i), (dx + 10, dy - 10),
264                              cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 2)
265              footer = "L=add  R/u=undo  r=reset  q=quit"
266              cv2.rectangle(view, (0, view.shape[0]-30), (view.shape[1], view.shape[0]), (0,0,0), -1)
267              cv2.putText(view, footer, (10, view.shape[0]-8),
268                          cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255,255,255), 1)
269              cv2.imshow(window_title, view)
270  
271          def on_mouse(event, x, y, flags, param):
272              nonlocal pts
273              if event == cv2.EVENT_LBUTTONDOWN and len(pts) < 3:
274                  ox, oy = int(x / scale), int(y / scale)
275                  ox = max(0, min(w - 1, ox))
276                  oy = max(0, min(h - 1, oy))
277                  pts.append((ox, oy))
278                  redraw()
279              elif event == cv2.EVENT_RBUTTONDOWN and pts:
280                  pts.pop()
281                  redraw()
282  
283          cv2.namedWindow(window_title, cv2.WINDOW_AUTOSIZE)
284          cv2.setMouseCallback(window_title, on_mouse)
285          redraw()
286  
287          while True:
288              k = cv2.waitKey(20) & 0xFF
289              if k == ord('q'):
290                  cv2.destroyWindow(window_title)
291                  return None  # signal early-quit
292              elif k == ord('u') and pts:
293                  pts.pop(); redraw()
294              elif k == ord('r'):
295                  pts = []; redraw()
296              if len(pts) == 3:
297                  cv2.destroyWindow(window_title)
298                  return pts
299  
300      # ---- iterate over images ----
301      for i, path in enumerate(paths, 1):
302          img = cv2.imread(path)
303          if img is None:
304              print("Skipping unreadable:", path)
305              continue
306  
307          title = f"[{i}/{len(paths)}] Pick 3 points: {os.path.basename(path)}"
308          pts = pick_points(img, title)
309          if pts is None:  # user pressed 'q'
310              break
311  
312          h, w = img.shape[:2]
313          annotations.append({
314              "filename": os.path.basename(path),
315              "width": w,
316              "height": h,
317              "points": [{"x": int(x), "y": int(y)} for (x, y) in pts]
318          })
319  
320      # ---- save JSON ----
321      out_json = os.path.join(heatmaps_dir, json_name)
322      with open(out_json, "w", encoding="utf-8") as f:
323          json.dump({"images": annotations}, f, ensure_ascii=False, indent=2)
324  
325      print(f"Saved {len(annotations)} annotated image(s) â†’ {out_json}")

404  def _scale_points(points_xy: List[Tuple[int,int]],
405                    src_wh: Tuple[int,int] | None,
406                    dst_hw: Tuple[int,int]) -> List[Tuple[int,int]]:
407      if not src_wh or src_wh[0] <= 0 or src_wh[1] <= 0:
408          return [(int(x), int(y)) for (x, y) in points_xy]
409      json_w, json_h = src_wh
410      H, W = dst_hw
411      sx, sy = W / float(json_w), H / float(json_h)
412      pts = [(int(round(x * sx)), int(round(y * sy))) for (x, y) in points_xy]
413      return [(min(max(0, x), W - 1), min(max(0, y), H - 1)) for (x, y) in pts]

415  def _radial_field(shape_hw: Tuple[int,int], center_xy: Tuple[int,int],
416                    R: float, gamma: float) -> np.ndarray:
417      H, W = shape_hw
418      x, y = center_xy
419      yy, xx = np.ogrid[:H, :W]
420      r = np.sqrt((yy - y)**2 + (xx - x)**2)
421      with np.errstate(invalid='ignore'):
422          h = 1.0 - np.power(r / max(R, 1e-6), gamma)
423      h[r > R] = 0.0
424      return np.clip(h, 0.0, 1.0).astype(np.float32)

426  def _combine(fields: List[np.ndarray], mode: str) -> np.ndarray:
427      if not fields:
428          return None
429      stack = np.stack(fields, axis=0)
430      if mode == "max":
431          out = np.max(stack, axis=0)
432      elif mode == "sum":
433          out = np.sum(stack, axis=0)
434          m = out.max()
435          if m > 0:
436              out /= m
437      return out.astype(np.float32)

518  def _extract_patch(img: np.ndarray, center_xy: Tuple[int, int], size: int) -> np.ndarray:
519      """Return a (size x size) RGB crop centered at (x,y). Assumes border checks upstream."""
520      x, y = center_xy
521      r = size // 2
522      return img[y - r:y + r, x - r:x + r].copy()

524  def _l2(a: Tuple[int,int], b: Tuple[int,int]) -> float:
525      ax, ay = a; bx, by = b
526      return float(np.hypot(ax - bx, ay - by))

528  def _coords_from_mask(mask: np.ndarray) -> List[Tuple[int,int]]:
529      """Convert HxW boolean mask to list of (x,y) coordinates."""
530      ys, xs = np.where(mask)
531      return list(zip(xs.tolist(), ys.tolist()))

533  def _pick_centers(mask_coords: List[Tuple[int,int]], how_many: int, min_dist: int) -> List[Tuple[int,int]]:
534      """Greedy selection of well-spaced centers (randomized order, L2 spacing)."""
535      chosen: List[Tuple[int,int]] = []
536      random.shuffle(mask_coords)
537      for (x, y) in mask_coords:
538          if all(_l2((x, y), c) >= min_dist for c in chosen):
539              chosen.append((x, y))
540              if len(chosen) >= how_many:
541                  break
542      return chosen
543  
544  def _compute_ring_index(x: int, y: int, centers_xy: List[Tuple[int,int]], R: int, n_bins: int) -> int:
545      """Assign a patch to a radial ring bin (0...n_bins-1) based on distance to nearest JSON center."""
546      if not centers_xy:
547          return n_bins - 1
548      r_min = min(_l2((x, y), (cx, cy)) for (cx, cy) in centers_xy)
549      r_norm = np.clip(r_min / float(max(R, 1e-6)), 0.0, 0.999)  # keep in [0,1)
550      return int(np.floor(r_norm * n_bins))

552  # ---- Main sampling routine ----
553  def _sample_patches_for_image(rec: Dict) -> List[Dict]:
554      """Produce rows of metadata for the HOT/COLD patches of a single image record."""
555      image_id = rec["image_id"]
556  
557      # Require cached heatmap from Cell 4
558      if image_id not in heatmaps:
559          # Fail fast to avoid silent re-computation and code duplication
560          raise RuntimeError(f"Missing cached heatmap for image_id={image_id}. Run Cell 4 first.")
561  
562      heat_syn = heatmaps[image_id]  # float32 [0,1], shape HÃ—W
563      orig_bgr = cv2.imread(rec["reference_file"])
564      if orig_bgr is None:
565          return []
566  
567      orig = cv2.cvtColor(orig_bgr, cv2.COLOR_BGR2RGB)
568      H, W = orig.shape[:2]
569  
570      # Safety: ensure heatmap matches the image size
571      if heat_syn.shape != (H, W):
572          heat_syn = cv2.resize(heat_syn, (W, H), interpolation=cv2.INTER_LINEAR).astype(np.float32)
573          heat_syn = np.clip(heat_syn, 0.0, 1.0)
574  
575      # Safe border mask (avoid crops crossing the image boundary)
576      yy, xx = np.ogrid[:H, :W]
577      safe_border = (xx >= BORDER_SAFE) & (xx < W - BORDER_SAFE) & (yy >= BORDER_SAFE) & (yy < H - BORDER_SAFE)
578  
579      # HOT / COLD masks
580      hot_mask  = (heat_syn >= HOT_THRESH)  & safe_border
581      cold_mask = (heat_syn <= COLD_THRESH) & safe_border
582  
583      hot_coords  = _coords_from_mask(hot_mask)
584      cold_coords = _coords_from_mask(cold_mask)
585  
586      # Fallback: relax thresholds if too few candidates
587      if len(hot_coords) < N_HOT_PER_IMG // 2:
588          hot_coords = _coords_from_mask((heat_syn >= HOT_THRESH * 0.8) & safe_border)
589      if len(cold_coords) < N_COLD_PER_IMG // 2:
590          cold_coords = _coords_from_mask((heat_syn <= COLD_THRESH * 1.2) & safe_border)
591  
592      # Pick well-spaced centers
593      hot_centers  = _pick_centers(hot_coords,  N_HOT_PER_IMG,  MIN_CENTER_DIST)
594      cold_centers = _pick_centers(cold_coords, N_COLD_PER_IMG, MIN_CENTER_DIST)
595  
596      # Crop and save patches; collect rows
597      rows: List[Dict] = []
598      pts_img = rec.get("points", [])  # for ring metadata only (JSON points in image coords if you used scale earlier)
599      for kind, centers in (("hot", hot_centers), ("cold", cold_centers)):
600          for i, (x, y) in enumerate(centers):
601              patch = _extract_patch(orig, (x, y), PATCH_SIZE)
602              if patch.shape[0] != PATCH_SIZE or patch.shape[1] != PATCH_SIZE:
603                  continue  # skip near borders just in case
604  
605              base = f"{image_id}_{kind}_{i:03d}_x{x}_y{y}"
606              f_po = ORIG_PATCH_DIR / f"{base}.png"
607              cv2.imwrite(str(f_po), cv2.cvtColor(patch, cv2.COLOR_RGB2BGR))
608  
609              rows.append({
610                  "image_id": image_id,
611                  "kind": kind,                  # "hot" | "cold"
612                  "x": x, "y": y,                # patch center
613                  "h_c": float(heat_syn[y, x]),  # heat value at center
614                  "ring": _compute_ring_index(x, y, pts_img, R=RADIUS_PX, n_bins=N_BINS),
615                  "patch_size": PATCH_SIZE,
616                  "orig_path": str(f_po),
617                  "radius_px": RADIUS_PX,
618                  "gamma_decay": GAMMA_DECAY,
619                  "combine_mode": COMB_MODE,
620                  "codec": rec.get("codec", "")
621              })
622      return rows

670  def _read_rgb(path: str) -> np.ndarray:
671      bgr = cv2.imread(path)
672      if bgr is None:

674      return cv2.cvtColor(bgr, cv2.COLOR_BGR2RGB)

677      def _extract_patch(img: np.ndarray, center_xy: Tuple[int,int], size: int) -> np.ndarray:
678          x, y = center_xy

682  def _blend_inner_patch(orig: np.ndarray, comp: np.ndarray, alpha: float, inner_size: int) -> np.ndarray:
683      """Blend ONLY the centered inner square (inner_sizeÃ—inner_size)."""
684      assert orig.shape == comp.shape, "orig/comp must share shape"
685      H, W = orig.shape[:2]
686      cx, cy = W // 2, H // 2
687      r = inner_size // 2
688      x0, y0 = max(0, cx - r), max(0, cy - r)
689      x1, y1 = min(W, cx + r), min(H, cy + r)
690      out = orig.copy().astype(np.float32)
691      roi = alpha * orig[y0:y1, x0:x1].astype(np.float32) + (1.0 - alpha) * comp[y0:y1, x0:x1].astype(np.float32)
692      out[y0:y1, x0:x1] = roi
693      return np.clip(out, 0, 255).astype(np.uint8)
